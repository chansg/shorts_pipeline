# ===== Gameplay pipeline extras (heavy / GPU-specific) =====
# Kept separate from requirements.txt so a lore-only install stays small.
# Install on top of the base requirements:
#     pip install -r requirements.txt
#     pip install -r requirements-gameplay.txt
#
# IMPORTANT — torch must match your CUDA. WhisperX pulls torch, but the default
# wheel may be CPU-only. For an NVIDIA GPU (e.g. RTX 3080), install the CUDA
# build FIRST, from PyTorch's index, then this file:
#     pip install torch torchaudio --index-url https://download.pytorch.org/whl/cu121
#     pip install -r requirements-gameplay.txt
# (cu121 = CUDA 12.1; pick the tag matching your driver — see pytorch.org.)

# huggingface-hub: whisperx 3.8.6 + its transformers + pyannote.audio all require
# huggingface-hub<1.0, but gradio 6.x DECLARES huggingface-hub>=1.2 (it runs fine on
# <1.0 in practice). Without this pin, `pip install -r requirements.txt` bumps the hub
# to satisfy gradio and transcription dies with:
#   "ImportError: huggingface-hub>=0.34.0,<1.0 is required ... found 1.x".
# Pin it low so transcription keeps working. pip WILL print a dependency-conflict
# warning about gradio — that is expected and safe to ignore (gradio still works).
# Already broke it? Recover with:  pip install "huggingface-hub<1.0"
huggingface-hub>=0.34.0,<1.0

# Transcribe + word-align + diarize (one pass). PINNED to the tested combo — the
# whisperx 3.2+ API moved DiarizationPipeline to whisperx.diarize and changed the
# ctor kwarg (use_auth_token -> token); the diarization code targets this pair.
whisperx==3.8.6

# Speaker diarization model. Needs a free HuggingFace token (HF_TOKEN in .env)
# AND one-time acceptance of the pyannote model licences on huggingface.co (a very
# common gotcha — diarization fails until you click "Agree" on BOTH):
#   - https://huggingface.co/pyannote/speaker-diarization-community-1   (default in whisperx 3.8)
#   - https://huggingface.co/pyannote/segmentation-3.0
pyannote.audio==4.0.4

# torch/torchaudio: PIN the version, but the wheel MUST be the CUDA build — install
# it FIRST from the CUDA index (see the note above), otherwise you get the CPU-only
# build and WhisperX runs on CPU (slow; full-auto becomes impractical).
torch==2.8.0
torchaudio==2.8.0
